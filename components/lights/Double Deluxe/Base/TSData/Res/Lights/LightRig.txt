# Light rig parser commands:
#
# lightrig <string>
# Start a light rig definition with the specified logical name
#
# autoRig [<state:string>]
# Turns on auto-lighting (the lighting manager).
#
# resource <string>
# Specifies the on-disk name for the light rig scene graph resource
#
# root <string>
# Specifies the node tag used to search for the root locator node in the light rig hierarchy
#


version 1


# Default lot lighting setup
lightrig Lot
   autoRig
end

# Neighbourhood
